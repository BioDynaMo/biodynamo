## Building ParaView for BioDynaMo ##
#####################################

## Step 1: Install QT 5.9.1

(LINUX)
wget https://download.qt.io/official_releases/qt/5.9/5.9.1/qt-opensource-linux-x64-5.9.1.run
chmod +x qt-opensource-linux-x64-5.9.1.run
./qt-opensource-linux-x64-5.9.1.run
"follow the instruction in the GUI"

(MAC OS)
wget https://download.qt.io/official_releases/qt/5.9/5.9.1/qt-opensource-mac-x64-5.9.1.dmg
hdiutil attach qt-opensource-mac-x64-5.9.1.dmg
open /Volumes/qt-opensource-mac-x64-5.9.1/qt-opensource-mac-x64-5.9.1.app
"follow the instruction in the GUI"
hdiutil detach /Volumes/qt-opensource-mac-x64-5.9.1


## Step 2: Clone paraview github repository
git clone https://gitlab.kitware.com/paraview/paraview.git
cd paraview
git submodule update --init --recursive
git checkout v5.4.0
git pull
git submodule update

## Step 3: Generate the cmake files for paraview
#
# If you want to build against a specific Qt library at /path/to/qt/cmake
# then set CMAKE_PREFIX_PATH=/path/to/qt/cmake (in this dir all the Qt modules
# should be available)

mkdir ../paraview-build
mkdir ../paraview-install
cd ../paraview-build

# The CMAKE_INSTALL_RPATH will put all the specified paths in all the installed
# targets (libraries and binaries) (upon make install). Since the relative paths
# from the ParaView targets are always the same we can set the rpaths to be
# relative from the ParaView targets (which are located at @loader_path). This
# makes the ParaView installation portable (as long as we copy Qt with it)

$cmake \
-DCMAKE_INSTALL_PREFIX:STRING="../paraview-install" \
-DCMAKE_BUILD_TYPE:STRING="Release" \
-DPARAVIEW_ENABLE_PYTHON:BOOL=ON \
-DPARAVIEW_ENABLE_MPI:BOOL=OFF \
-DPARAVIEW_INSTALL_DEVELOPMENT_FILES:BOOL=ON \
-DCMAKE_MACOSX_RPATH:BOOL=ON \
-DCMAKE_INSTALL_RPATH:STRING="@loader_path/../../qt/lib;@loader_path/../../../../../qt/lib" \
 ../paraview

## Step 4:
make -j4
sudo make install

## Step 5:
cd <biodynamo-github-dir>/paraview_plugin
cd bdm_glyph ("or any other plugin you want to include")
mkdir build && cd build
cmake .. && make

(Linux)
cp libBDMGlyphFilter.so <paraview-install-dir/lib/paraview-5.4/plugins

(Mac OS)
cp libBDMGlyphFilter.dylib <paraview-install-dir/bin/paraview.app/Contents/MacOS/plugins
